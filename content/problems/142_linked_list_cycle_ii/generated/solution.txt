# Linked List Cycle II Solution

## Approach: Floyd's Cycle Detection Algorithm

The key insight is using Floyd's tortoise and hare algorithm in two phases:

1. **Phase 1**: Detect if a cycle exists using slow/fast pointers
2. **Phase 2**: Find the exact node where the cycle begins

### Mathematical Proof

Let:
- `F` = distance from head to cycle start
- `C` = cycle length  
- `a` = distance from cycle start to meeting point

When slow and fast pointers meet:
- Slow traveled: `F + a`
- Fast traveled: `F + a + n*C` (n complete cycles)
- Since fast = 2 × slow: `2(F + a) = F + a + n*C`
- Therefore: `F = n*C - a`

This means: distance from **head to cycle start** = distance from **meeting point to cycle start**

## Code

```python
# Definition for singly-linked list.
class ListNode:
    def __init__(self, x):
        self.val = x
        self.next = None


class Solution:
    def detectCycle(self, head: ListNode) -> ListNode:
        """
        Detects the node where cycle begins using Floyd's algorithm.
        Time Complexity: O(n)
        Space Complexity: O(1)
        """
        if not head or not head.next:
            return None
        
        # Phase 1: Detect cycle using slow and fast pointers
        slow = head
        fast = head
        
        while fast and fast.next:
            slow = slow.next          # Move 1 step
            fast = fast.next.next     # Move 2 steps
            
            if slow == fast:          # Cycle detected
                break
        else:
            return None               # No cycle found
        
        # Phase 2: Find cycle start
        # Move one pointer to head, keep other at meeting point
        # Both move 1 step at a time - they meet at cycle start
        pointer1 = head
        pointer2 = slow               # Meeting point
        
        while pointer1 != pointer2:
            pointer1 = pointer1.next
            pointer2 = pointer2.next
        
        return pointer1               # Cycle start node


# ==================== Helper Functions ====================

def create_linked_list_with_cycle(values: list, pos: int) -> ListNode:
    """Creates a linked list with a cycle at position pos (-1 for no cycle)."""
    if not values:
        return None
    
    nodes = [ListNode(val) for val in values]
    
    # Link nodes
    for i in range(len(nodes) - 1):
        nodes[i].next = nodes[i + 1]
    
    # Create cycle if pos is valid
    if pos != -1:
        nodes[-1].next = nodes[pos]  # Tail connects to node at pos
    
    return nodes[0]


def get_cycle_start_index(head: ListNode) -> int:
    """Returns the index of cycle start node, or -1 if no cycle."""
    if not head:
        return -1
    
    visited = {}
    current = head
    index = 0
    
    while current:
        if id(current) in visited:
            return visited[id(current)]
        visited[id(current)] = index
        current = current.next
        index += 1
    
    return -1


# ==================== Test Cases ====================

def run_tests():
    solution = Solution()
    
    test_cases = [
        ([3, 2, 0, -4], 1, "tail connects to node index 1"),
        ([1, 2],         0, "tail connects to node index 0"),
        ([1],           -1, "no cycle"),
        ([],            -1, "no cycle"),
    ]
    
    print("=" * 60)
    print("Testing Linked List Cycle II")
    print("=" * 60)
    
    for values, pos, expected_output in test_cases:
        head = create_linked_list_with_cycle(values, pos)
        result = solution.detectCycle(head)
        
        if result:
            cycle_index = get_cycle_start_index(head)
            actual_output = f"tail connects to node index {cycle_index}"
        else:
            actual_output = "no cycle"
        
        status = "✅ PASS" if actual_output == expected_output else "❌ FAIL"
        
        print(f"\nInput:    head = {values}, pos = {pos}")
        print(f"Expected: {expected_output}")
        print(f"Got:      {actual_output}")
        print(f"Status:   {status}")
    
    print("\n" + "=" * 60)


if __name__ == "__main__":
    run_tests()
```

## Complexity Analysis

| | Complexity |
|---|---|
| **Time** | O(n) - each node visited at most twice |
| **Space** | O(1) - only two pointers used |
